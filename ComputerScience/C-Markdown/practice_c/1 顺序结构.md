#### 1.1 输入三个数并输出其中的最小值

```c
#include<stdio.h>
int main()
{
	int a,b,c;
	int temp;
	printf("请输入三个整数:\n");
	scanf("%d%d%d",&a,&b,&c);
	int min=0;
	//始终让a小于b和c
	if(a>b){
		temp = a;
		a=b;
		b=temp;
	}
	if(a>c){
		temp = a;
		a=c;
		c=temp;
	}
	if(b>c){
		temp=b;
		b=c;
		c=temp;
	}
	min = a;
	printf("最小值是：%d",min);
	
	return 0;
}
```



#### 1.2 随机产生一个3位数的自然数


#### 1.3报告读取的空格数、换行符数和所有其他字符的数量，读到#字符停止，

```c
#include<stdio.h>

int main()
{
	char ch;
	int isspace_n = 0;
	int ishuanhang = 0;
	int char_n = 0;
	printf("请输入字符：");
	while ( (ch = getchar()) != '#') {
		if (ch == ' ') {
			isspace_n++;
		}else if (ch == '\n') {
			ishuanhang++;
		}
		else {
			char_n++;
		}
	}
	printf("其他字符数：%d，空格数：%d,换行数：%d", char_n, isspace_n, ishuanhang);
}
```
